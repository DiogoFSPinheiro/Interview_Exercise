25/**
26* @brief Removes and returns an element from the queue with a timeout.
27*
28* Blocks the calling thread until an item is available or the specified timeout is reached.
29*
30* @param milliseconds The timeout in milliseconds to wait for an element.
31* @return The item popped from the queue.
32*
33* @throws std::runtime_error If no element is received within the given timeout.
34*/
35template<typename T>
36T Queue<T>::PopWithTimeout(int milliseconds)
37{
38 std::unique_lock<std::mutex> lock(mtx);
39
40 if (_count == 0)
41 {
42 if (!cv.wait_for(lock, std::chrono::milliseconds(milliseconds), [this] { return _count > 0; }))
43 {
44 throw std::runtime_error("No element received in time.");
45 }
46 }
47
48 T value = _buffer[_head];
49 advance(_head);
50 --_count;
51
52 return value;
53}

55/**
56* @brief Removes and returns an element from the queue.
57*
58* Blocks the calling thread until an item becomes available.
59*
60* @return The item popped from the queue.
61*/
62template<typename T>
63T Queue<T>::Pop()
64{
65 std::unique_lock<std::mutex> lock(mtx);
66
67 cv.wait(lock, [this]() { return _count > 0; });
68
69 T value = _buffer[_head];
70 advance(_head);
71 --_count;
72
73 return value;
74}

76/**
77* @brief Adds an element to the queue.
78*
79* If the queue is full, it discards the oldest element before inserting the new one.
80* Notifies one waiting thread (if any).
81*
82* @param element The element to insert.
83*/
84template <typename T>
85void Queue<T>::Push(T element)
86{
87 std::unique_lock<std::mutex> lock(mtx);
88
89 if (_count == _maxSize)
90 {
91 advance(_head);
92 --_count;
93 }
94
95 _buffer[_tail] = element;
96 advance(_tail);
97 ++_count;
98
99 cv.notify_one();
100}
